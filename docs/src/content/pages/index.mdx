```astro live
---
import { Button, Card, Deck, Heading, Image, Section, Spread } from 'fulldev-ui'

const cards = Array.from({ length: 20 })
---

<Section size="sm">
  <Spread>
    <Heading> Here are some products </Heading>
    <Button> All products </Button>
  </Spread>
  <Deck structure="carousel" frame="fill">
    {
      cards.map((card) => (
        <Card frame="none" size="sm">
          <Image src="/example.jpg" />
          <Heading>Product</Heading>
        </Card>
      ))
    }
  </Deck>
</Section>
```
