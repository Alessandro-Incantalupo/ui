import { Tabs, TabItem, LinkCard, Card } from '@astrojs/starlight/components';

Run the install command.

<Tabs>
  <TabItem label="pnpm">
    ```bash
    pnpm add @fullui/ui
    ```
  </TabItem>
  <TabItem label="yarn">
    ```bash
    yarn add @fullui/ui
    ```
  </TabItem>
  <TabItem label="npm">
    ```bash
    npm install @fullui/ui
    ```
  </TabItem>
  <TabItem label="bun">
    ```bash
    bun i @fullui/ui
    ```
</TabItem>
</Tabs>
<br/>
That's it! Just import the components you need, no extra configuration required.
