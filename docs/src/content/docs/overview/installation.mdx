import { Tabs, TabItem, LinkCard, Card } from '@astrojs/starlight/components';

All components work on their own, no configuration or integration needed. 

Just run the install command below and import only the components you need.

<Tabs>
  <TabItem label="pnpm">
    ```bash
    pnpm add @fullui/ui
    ```
  </TabItem>
  <TabItem label="yarn">
    ```bash
    yarn add @fullui/ui
    ```
  </TabItem>
  <TabItem label="npm">
    ```bash
    npm install @fullui/ui
    ```
  </TabItem>
</Tabs>
