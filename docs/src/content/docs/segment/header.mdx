---
title: Header
description: Component is used to display a header at the top of the page.
props:
  - name: structure
    type: spread | row | column
    default: spread
  - name: position
    type: relative | sticky | fixed | absolute
    default: relative
  - name: frame
    type: fill | panel
    default: fill
  - name: size
    type: sm | md | lg
  - name: compact
    type: attribute
    default: true
  - name: align
    type: start | center | end
    default: center
  - name: variant
    type: ghost | subtle | soft | surface | outline
    default: subtle
  - name: theme
    type: string
  - name: as
    type: HTMLTag
    default: header
  - name: HTML Attributes
    type: Polymorphic<as>
