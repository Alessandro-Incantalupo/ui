title: Header

props:
  - name: structure
    type: spread | row | column
  - name: position
    type: 'relative | sticky | fixed | absolute'
  - name: frame
    type: fill | panel
  - name: size
    type: 'sm | md | lg'
  - name: compact
    type: attribute
  - name: align
    type: 'start | center | end'
  - name: variant
    type: ghost | raised
    default: subtle
  - name: theme
    type: string
  - name: as
    type: HTMLTag
    default: header
  - name: HTML Attributes
    type: Polymorphic<as>
