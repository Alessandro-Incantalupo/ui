---
title: Footer
description: Component is used to display a footer at the bottom of the page.
props:
  - name: structure
    type: spread | row | split | column
  - name: frame
    type: fill | panel
  - name: size
    type: sm | md | lg
  - name: compact
    type: attribute
    default: true
  - name: align
    type: start | center | end
    default: center
  - name: variant
    type: ghost | subtle | soft | surface
  - name: theme
    type: string
  - name: as
    type: HTMLTag
    default: footer
  - name: HTML Attributes
    type: Polymorphic<as>
---

import PropsTable from '../../../components/PropsTable.astro'

---
import { Column, Footer, Heading, Link, Paragraph } from 'fulldev-ui'
---

  <Column align="start">
    <Heading as="a" href="/">Fulldev UI</Heading>
    <Paragraph>Rapidly build content-driven websites.</Paragraph>
  </Column>
  <Column>
    <Link href="/overview/introduction">Docs</Link>
    <Link href="/components">Components</Link>

  </Column>
</Footer>
