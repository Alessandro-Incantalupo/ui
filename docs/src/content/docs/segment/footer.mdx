    type: boolean
    default: true
  - name: align
    type: start | center | end
    default: center
  - name: variant
    type: ghost | subtle | soft | surface
    default: subtle
  - name: theme
    type: string
  - name: as
    type: HTMLTag
    default: footer
  - name: HTML Attributes
    type: Polymorphic<as>
