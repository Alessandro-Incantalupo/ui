---
title: Footer
description: Component is used to display a footer at the bottom of the page.
props:
  - name: structure
    type: spread
  - name: position
    type: relative | sticky | fixed | absolute
  - name: frame
    type: fill
  - name: size
    type: sm | md | lg
  - name: compact
    type: attribute
    default: true
  - name: align
    type: start | center | end
    default: center
  - name: variant
    type: ghost | subtle | soft | surface
  - name: theme
    type: string
  - name: as
    type: HTMLTag
    default: header
  - name: HTML Attributes
    type: Polymorphic<as>
---

import PropsTable from '../../../components/PropsTable.astro'
