## Props
Below props are available in addition to all [Tag](/foundation/tag) props.

```astro live
---
import { Select } from '@fullui/ui'

const options = [
  {
    value: 'option 1',
    text: 'Option 1'
  },
  {
    value: 'option 2',
    text: 'Option 2'
  }
]
---

<Select name="name" placeholder="Name" {options} />
```

### `type`
```astro live
---
import { Input } from '@fullui/ui'
---

<Input name="email" placeholder="email" type="email" />
<Input name="password" placeholder="password" type="password" />
<Input name="number" placeholder="number" type="number" />
```

### `size`
**Type:** `variant?: 'sm' | 'md' | 'lg'`
```astro live
---
import { Select } from '@fullui/ui'

const options = [
  {
    value: 'option 1',
    text: 'Option 1'
  },
  {
    value: 'option 2',
    text: 'Option 2'
  }
]
---

<Select size="sm" {options}/>
<Select size="md" {options}/>
<Select size="lg" {options}/>
```

### `compact`
**Type:** `compact?: boolean`
```astro live
---
import { Select } from '@fullui/ui'

const options = [
  {
    value: 'option 1',
    text: 'Option 1'
  },
  {
    value: 'option 2',
    text: 'Option 2'
  },
]
---

<Select density="compact" size="sm" {options} />
<Select density="compact" size="md" {options} />
<Select density="compact" size="lg" {options} />
```

```astro live
import { Select, Option } from '@fullui/ui'

  },

  {

  }

```
