Astro form component used for creating a form.

## Props
extends [Tag](/utility/tag) and [Children](/utility/children).

## Defaults
```ts
as = 'form'
method = 'POST'
```

## Example
The example below uses regular syntax, check out [Syntax](/overview/syntax) for more options.

### Input
```astro live
---
import { Form, Field, Label, Input, Textarea, Select, Checkbox, Radio, Button } from '@fullui/ui'
---

<Form action="/">
  <Field>
    <Label>Input</Label>
    <Input name="password" placeholder="Password" type="password" />
  </Field>
  <Field>
    <Label>Textarea</Label>
    <Textarea name="description" placeholder="Description" rows="6" />
  </Field>
  <Field checkbox>
    <Checkbox name="checkbox" />
    <Label>Checkbox</Label>
  </Field>
  <Field>
    <Label>Select</Label>
    <Select>
      <option value="1">Option 1</option>
      <option value="2">Option 2</option>
      <option value="3">Option 3</option>
    </Select>
  </Field>
  <Field radio>
    <Radio name="radio" />
    <Label>Radio 1</Label>
  </Field>
  <Field radio>
    <Radio name="radio" />
    <Label>Radio 2</Label>
  </Field>
  <Field radio>
    <Radio name="radio" />
    <Label>Radio 3</Label>
  </Field>
  <Button as="button" type="submit" color="base" variant="surface">Submit</Button>
</Form>
```
