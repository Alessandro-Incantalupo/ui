### Props
Extends [Tag](/utility/tag), [Children](/utility/children)
```ts
checkbox?: any
radio?: any
```

### Defaults
```ts
as = 'input'
```

## Examples
The examples below could also be written with [different syntax](/overview/syntax).

### Input
```astro live
---
import { Field, Label, Input } from '@fullui/ui'
---

<Field>
  <Label>Input</Label>
  <Input name="password" placeholder="Password" type="password" />
</Field>
```

### Textarea
```astro live
---
import { Field, Label, Textarea } from '@fullui/ui'
---

<Field>
  <Label>Textarea</Label>
  <Textarea name="description" placeholder="Description" rows="6" />
</Field>
```

### Select
```astro live
---
import { Field, Label, Select } from '@fullui/ui'
---

<Field>
  <Label>Select</Label>
  <Select>
    <option value="1">Option 1</option>
    <option value="2">Option 2</option>
    <option value="3">Option 3</option>
  </Select>
</Field>
```

### Checkbox
```astro live
---
import { Field, Label, Checkbox } from '@fullui/ui'

```

### Radio
