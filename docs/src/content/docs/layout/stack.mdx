Astro stack component used for alignment and direction.

### Props
extends [Tag](/utility/tag) and [Children](/utility/children).
```ts
direction?: 'column' | 'row'
align?: 'start' | 'center'
```

### Defaults
```ts
direction = 'column'
```

## Examples
The example below uses regular syntax, check out [Syntax](/overview/components) for more options.

### Columns
```astro live
---
import { Stack, DecorativeBox } from '@fullui/ui'
---

<Stack align="center" direction="column">
  <DecorativeBox width="45%" height="300px" />
  <DecorativeBox width="45%" height="300px" />
</Stack>
```

### Rows
```astro live
---
import { Stack, DecorativeBox } from '@fullui/ui'
---

<Stack align="start" direction="row">
  <DecorativeBox width="45%" height="300px" />
  <DecorativeBox width="45%" height="300px" />
</Stack>
```
