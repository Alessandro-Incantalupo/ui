Astro stack component used for alignment and direction.

## Props
extends [Tag](/utility/tag) and [Children](/utility/children).
```ts
direction?: 'column' | 'row'
align?: 'start' | 'center'
```

## Examples
Here are some examples of ways you can use this component.

### Columns
```astro live
---
import { Stack } from '@fullui/ui'
import { DecorativeBox } from '@fullui/ui'
---

<Stack align="center" direction="column">
  <DecorativeBox width="45%" height="300px" />
  <DecorativeBox width="45%" height="300px" />
</Stack>
```

### Rows
```astro live
---
import { Stack } from '@fullui/ui'
import { DecorativeBox } from '@fullui/ui'
---

<Stack align="start" direction="row">
  <DecorativeBox width="45%" height="300px" />
  <DecorativeBox width="45%" height="300px" />
</Stack>
```
