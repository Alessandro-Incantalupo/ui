Create an input field for use in a form.

## Props
extends [Endomorphic](/guides/components), [Tag](/components/tag), [Label](/components/label), [Options](/components/options), [Checkbox](/components/checkbox), [Textarea](/components/textarea) and [Input](/components/input).
```ts
interface Props
  extends Endomorphic<Props, 'field'>,
    ComponentProps<typeof Tag>,
    Pick<ComponentProps<typeof Label>, 'label' | '_label'>,
    Pick<ComponentProps<typeof Options>, 'options' | '_options'>,
    Pick<ComponentProps<typeof Checkbox>, 'checkbox' | '_checkbox'>,
    Pick<ComponentProps<typeof Textarea>, 'textarea' | '_textarea'>,
    Pick<ComponentProps<typeof Input>, 'input' | '_input'> {}
```

## Examples
```astro live
---
import { Field } from '@fulldevlabs/fullui'
---

<Field 
  label="Subject"
  input="subject" 
  placeholder="Subject" 
  type="text"
/>
```
