## Props
extends [Tag](/components/tag) and [Card](/components/card).
```ts
layout: 'carousel' | 'grid' | 'masonry' | 'row' | 'panels' | 'bento' | null | undefined
```

## Example
```astro live
---
import { CardGroup } from '@fulldevlabs/fullui'

const myCards = [
  {
    variant: "solid",
    image: {src: "https://picsum.photos/id/237/536/354"},
    composite: {
      display: {
        writeup: {
          title: {html: "This is a card"},
          description: {
            html: "Lorem ipsum dolor sit amet, consectetur adipiscing elit."
          },
        }
      }
    }
  },
  {
    variant: "solid",
    image: {src: "https://picsum.photos/id/237/536/354"},
    composite: {
      display: {
        writeup: {
          title: {html: "This is also a card"},
          description: {
            html: "Lorem ipsum dolor sit amet, consectetur adipiscing elit."
          },
        }
      }
    }
  }
]
---

<CardGroup
  layout="grid"
  cards={myCards}
/>
```
