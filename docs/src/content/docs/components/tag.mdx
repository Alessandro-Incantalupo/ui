A smart component which can render as different html elements, aplies styling classes, and only renders when it actually has content.

This component is more of a helper allmost every component in Fullui uses this component. You might find yourself explicitly using it, yet it's good to know how it works.

## Props
```ts
type Poly<Rest extends {} = {}> = {
  as?: HTMLTag
} & Rest

interface Props extends Poly {
  if?: string | boolean

  class?: HTMLAttributes<HTMLTag>['class']
  id?: HTMLAttributes<HTMLTag>['id']

  compact?: boolean | null
  scheme?: 'light' | 'dark' | null
  color?: 'base' | 'accent' | null
  size?: '1' | '2' | '3' | '4' | null
  variant?: 'solid' | 'soft' | 'surface' | 'outline' | 'ghost' | null
  contrast?: boolean | null
}
```
