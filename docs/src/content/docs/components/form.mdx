## Props
extends [Tag](/components/tag), [FieldGroup](/components/field-group) and [Button](/components/button).
```ts
method: string | null | undefined
```

## Example
```astro live
---
import { Form } from '@fulldevlabs/fullui'
import { FieldGroup } from '@fulldevlabs/fullui'

const myFields = [
  {label: {html: "Email"}, input: {name: "email"}},
  {label: {html: "Subject"}, input: {name: "subject"}},
  {label: {html: "Message"}, textarea: {name: "message"}},
]
---

<Form
  method="POST"
  fieldGroup={{fields: myFields}}
  button={{
    html: "Submit", 
    size: "1", 
    variant: "surface", 
    icon: {name: "arrow-right"}, 
    contrast: true
  }}
/>
```
