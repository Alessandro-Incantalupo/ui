## Props
Extends [Tag](/components/tag) and [Icon](/components/icon).
```ts
href: string | null | undefined
variant: 'solid' | 'soft' | 'surface' | 'outline' | 'ghost' | null | undefined
```

## Defaults
```ts
as = 'a'
variant = 'solid'
color = 'accent'
```

## Examples
Below examples use regular props, but also accept [Endomorphic props](/guides/macromorphic).

### Variants
```astro live
---
import { Button } from '@fulldevlabs/fullui'
---

<Button variant="solid">Button</Button>
<Button variant="soft">Button</Button>
<Button variant="surface">Button</Button>
<Button variant="outline">Button</Button>
<Button variant="ghost">Button</Button>
```

### Color
```astro live
---
import { Button } from '@fulldevlabs/fullui'
---

<Button color="accent" variant="solid">Button</Button>
<Button color="accent" variant="soft">Button</Button>
<Button color="accent" variant="surface">Button</Button>
<Button color="accent" variant="outline">Button</Button>
<Button color="accent" variant="ghost">Button</Button>
```

### Contrast
```astro live
---
import { Button } from '@fulldevlabs/fullui'
---

<Button contrast variant="solid">Button</Button>
<Button contrast variant="soft">Button</Button>
<Button contrast variant="surface">Button</Button>
<Button contrast variant="outline">Button</Button>
<Button contrast variant="ghost">Button</Button>
```

### Icon
```astro live
---
import { Button } from '@fulldevlabs/fullui'
import { Icon } from '@fulldevlabs/fullui'
---

<Button icon={{name: "arrow-right"}} contrast></Button>

<Button icon={{name: "arrow-right"}} contrast>Button</Button>
```

### Size
```astro live
---
import { Button } from '@fulldevlabs/fullui'
---

<Button size="3" contrast>Button</Button>
<Button size="2" contrast>Button</Button>
<Button size="1" contrast>Button</Button>
```

### Compact
*It might seem weird having sizes and also compact, but it has a lot of benefits. More on this later*
```astro live
---
import { Button } from '@fulldevlabs/fullui'
---

<Button size="3" compact contrast>Button</Button>
<Button size="2" compact contrast>Button</Button>
<Button size="1" compact contrast>Button</Button>
```
