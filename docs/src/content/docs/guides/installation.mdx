import { Tabs, TabItem } from '@astrojs/starlight/components';

Run one of the following commands to install fullui into your project:

<Tabs>
  <TabItem label="pnpm">
    ```bash
    pnpm add @fulldevlabs/fullui
    ```
  </TabItem>
  <TabItem label="yarn">
    ```bash
    yarn add @fulldevlabs/fullui
    ```
  </TabItem>
  <TabItem label="npm">
    ```bash
    npm install @fulldevlabs/fullui
    ```
  </TabItem>
</Tabs>
